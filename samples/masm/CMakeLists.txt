set(SAMPLE_NAME sample-masm)

"-Fa${CMAKE_CURRENT_SOURCE_DIR}/code.asm -FA")

"-Fa${CMAKE_CURRENT_SOURCE_DIR}/main.asm -FA")

	NAME ${SAMPLE_NAME}
	COMMAND $<TARGET_FILE:${SAMPLE_NAME}>)
